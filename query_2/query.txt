QUERY GROUP BY:
Contare quanti iscritti ci sono stati ogni anno:
    SELECT COUNT(`id`) AS `students`,year(enrolment_date) AS `year` 
    FROM students
    GROUP BY `year`;

Contare gli insegnanti che hanno l'ufficio nello stesso edificio
    SELECT COUNT(`id`) AS `teachers`, `office_number` AS `office`
    FROM `teachers`
    GROUP BY `office`;

Calcolare la media dei voti di ogni appello d'esame
    SELECT AVG(`vote`) AS `media`, `exam_id` FROM `exam_student`
    GROUP BY `exam_id`;

Contare quanti corsi di laurea ci sono per ogni dipartimento
    SELECT COUNT(id) AS `numero_corsi`, department_id FROM degrees
    GROUP BY department_id;

QUERY JOINS:
Selezionare tutti gli studenti iscritti al Corso di Laurea in Economia
    SELECT `students`.`name` AS `studente`,
            `students`.`id`,
            `degrees`.`name` AS `corso`
    FROM `students`
    JOIN `degrees` 
    ON `students`.`degree_id`= `degrees`.`id`
    WHERE `degrees`.`name` = 'Corso di Laurea in Economia';

Selezionare tutti i Corsi di Laurea Magistrale del Dipartimento di Neuroscienze
SELECT `degrees`.`level` AS `livello`,
        `degrees`.`id`,
        `degrees`.`name`,
        `departments`.`name` AS `dipartimento`
FROM `degrees`
JOIN `departments`
ON `departments`.`id`= `degrees`.`department_id`
WHERE `departments`.`name`= 'Dipartimento di Neuroscienze' AND `degrees`.`level` = 'magistrale';


Selezionare tutti i corsi in cui insegna Fulvio Amato (id=44)
    SELECT `courses`.`name` AS `course_name`,
		`courses`.`id` AS `course_id`,
        `teachers`.`name` AS `teacher_name`,
		`teachers`.`id` AS `teacher_id`
    FROM `courses`
    JOIN `course_teacher`
    ON `courses`.`id`= `course_teacher`.`course_id`
    JOIN `teachers`
    ON `teachers`.`id`= `course_teacher`.`teacher_id`
    WHERE `teachers`.`id`= 44;

Selezionare tutti gli studenti con i dati relativi al corso di laurea a cui sono iscritti e il relativo dipartimento, in ordine alfabetico per cognome e nome
    SELECT `students`.`id` AS `student_id`,
        `students`.`name` AS `student_name`,
        `students`.`surname` AS `student_surname`,
        `degrees`.`id` AS `degree_id`,
        `degrees`.`name` AS `degrre_name`,
        `departments`.`id` AS `departments_id`,
        `departments`.`name` AS `departments_name`
    FROM `students`
    JOIN `degrees`
    ON `students`.`degree_id`= `degrees`.`id`
    JOIN `departments`
    ON `degrees`.`department_id` = `departments`.`id`
    ORDER BY `students`.`name`,`students`.`surname` DESC;

Selezionare tutti i corsi di laurea con i relativi corsi e insegnanti
    SELECT `degrees`.`name` AS `nome_corso_di_laurea`,
        `degrees`.`id` AS `id_corso_di_laurea`,
        `courses`.`name` AS `nome_corso`,
        `courses`.`id` AS `id_corso`,
        `teachers`.`name` AS `nome_insegnante`,
        `teachers`.`id` AS `id_insegnante`
    FROM `courses`
    JOIN `degrees`
    ON `courses`.`degree_id` = `degrees`.`id`
    JOIN `course_teacher`
    ON `course_id` = `courses`.`id`
    JOIN `teachers`
    ON `teacher_id` = `teachers`.`id`
    ORDER BY `degrees`.`name`;
